function RGB = SeparatetoSingleSignals(R,G,B)
% This function combines the separate color signals into a single 3-D signal

[rows, cols] = size(R);

RGB = zeros(rows,cols,3);

for i = 1:rows
    for j = 1:cols
        RGB(i,j,1) = R(i,j);
        RGB(i,j,2) = G(i,j);
        RGB(i,j,3) = B(i,j);
    end
end

return
