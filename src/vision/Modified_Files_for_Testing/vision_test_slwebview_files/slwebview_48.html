function [L,a,b]=CombineSegmentedImage(LabMatrix,ref_Colors)
% This function puts together the segmented imaged such that it can be shown

[rows, cols] = size(LabMatrix);
L = zeros(rows,cols);
a = zeros(rows,cols);
b = zeros(rows,cols);

for i = 1:rows
    for j = 1:cols
        L(i,j) = ref_Colors(LabMatrix(i,j),1);
        a(i,j) = ref_Colors(LabMatrix(i,j),2);
        b(i,j) = ref_Colors(LabMatrix(i,j),3);
    end
end

return
